# 用户端商城 - 前端架构设计

> 本文档为用户端商城前端的技术架构设计，融合 Shopify / Shopee / Amazon 的个性化与用户体验设计。

---

## 1. 技术选型

### 1.1 核心框架

| 层级 | 选项 | 说明 |
|------|------|------|
| 框架 | **React 18 + Vite** | 与 admin 保持一致 |
| UI 组件库 | **Ant Design 5.x** (PC) / **Ant Design Mobile 5.x** (Mobile) | 企业级组件 |
| 状态管理 | **Zustand** + **React Query** | 状态管理 + 服务端状态 |
| 路由 | **React Router 6** | 与 admin 保持一致 |
| HTTP 客户端 | **Axios** + **SWR** | 缓存与重新获取 |
| CSS 方案 | **Tailwind CSS** | 快速开发响应式 UI |

### 1.2 特色技术栈

| 用途 | 库 | 说明 |

| 虚拟列表 | react-virtualized | 长列表优化 |

### 3.2 购物车模块

```typescript
// features/cart/stores/cartStore.ts
interface CartState {
  items: CartItem[]
  selectedIds: string[]
  totalAmount: number
  totalCount: number
  addItem: (item: CartItem) => void
  removeItem: (id: string) => void
  updateQuantity: (id: string, quantity: number) => void
  toggleSelect: (id: string) => void
  selectAll: (selected: boolean) => void
  clearSelected: () => void
  syncToServer: () => Promise<void>
  moveToCollect: (ids: string[]) => Promise<void>
}

export const useCartStore = create<CartState>()(
  persist(
    (set, get) => ({
      items: [],
      selectedIds: [],

      addItem: (item) => {
        // 带动画效果
        animateAddToCart(item.id)
        set(state => {
          const existing = state.items.find(i => i.skuId === item.skuId)
          if (existing) {
            return {
              items: state.items.map(i =>
                i.skuId === item.skuId
                  ? { ...i, quantity: i.quantity + item.quantity }
                  : i
              )
            }
          }
          return { items: [...state.items, item] }
        })
        // 同步服务端
        get().syncToServer()
      },

      removeItem: (id) => set(state => ({
        items: state.items.filter(i => i.id !== id),
        selectedIds: state.selectedIds.filter(sid => sid !== id)
      })),

      updateQuantity: (id, quantity) => set(state => ({
        items: state.items.map(i => i.id === id ? { ...i, quantity } : i)
      })),

      toggleSelect: (id) => set(state => ({
        selectedIds: state.selectedIds.includes(id)
          ? state.selectedIds.filter(sid => sid !== id)
          : [...state.selectedIds, id]
      })),

      selectAll: (selected) => set(state => ({
        selectedIds: selected ? state.items.map(i => i.id) : []
      })),

      clearSelected: () => set({ selectedIds: [] }),

      syncToServer: async () => {
        const { items } = get()
        await syncCart(items)
      },

      moveToCollect: async (ids) => {
        // 移动到收藏夹
        await moveToFavorite(ids)
        get().removeItem(ids)
      }
    }),
    {
      name: 'cart-storage',
      partialize: (state) => ({ items: state.items })
    }
  )
)
```

### 3.3 订单模块

```typescript
// features/orders/hooks/useCheckout.ts
export const useCheckout = () => {
  const cartStore = useCartStore()

  const [addresses, setAddresses] = useState<Address[]>([])
  const [selectedAddress, setSelectedAddress] = useState<Address | null>(null)
  const [coupons, setCoupons] = useState<Coupon[]>([])
  const [selectedCoupon, setSelectedCoupon] = useState<Coupon | null>(null)
  const [remark, setRemark] = useState('')
  const [loading, setLoading] = useState(false)
  const [payTimeLeft, setPayTimeLeft] = useState(30 * 60) // 30分钟倒计时

  // 计算价格
  const { totalAmount, discountAmount, finalAmount } = useMemo(() => {
    const items = cartStore.items.filter(i => cartStore.selectedIds.includes(i.id))
    const total = items.reduce((sum, i) => sum + i.price * i.quantity, 0)
    const discount = selectedCoupon?.type === 'fixed'
      ? selectedCoupon.value
      : total * (selectedCoupon?.value / 100)
    return {
      totalAmount: total,
      discountAmount: discount,
      finalAmount: total - discount
    }
  }, [cartStore.items, cartStore.selectedIds, selectedCoupon])

  // 提交订单
  const submitOrder = async () => {
    setLoading(true)
    try {
      const order = await createOrder({
        addressId: selectedAddress!.id,
        couponId: selectedCoupon?.id,
        remark,
        items: cartStore.items.filter(i => cartStore.selectedIds.includes(i.id))
      })
      // 清除已购买的购物车商品
      cartStore.removeItem(cartStore.selectedIds)
      return order
    } finally {
      setLoading(false)
    }
  }

  // 支付倒计时
  useCountdown(payTimeLeft, (remaining) => {
    if (remaining === 0) {
      // 支付超时，订单自动取消
      message.warning('订单已超时取消')
    }
  })

  return {
    addresses, selectedAddress, setSelectedAddress,
    coupons, selectedCoupon, setSelectedCoupon,
    remark, setRemark,
    totalAmount, discountAmount, finalAmount,
    payTimeLeft,
    submitOrder, loading
  }
}
```

interface UserState {
  token: string | null
  userInfo: UserInfo | null
  isLoggedIn: boolean
  level: number
  points: number
  login: (username: string, password: string) => Promise<void>
  loginByCode: (phone: string, code: string) => Promise<void>
  socialLogin: (provider: 'wechat' | 'google' | 'apple') => Promise<void>
  logout: () => void
  updateProfile: (data: Partial<UserInfo>) => Promise<void>
  addPoints: (points: number, reason: string) => void
}

export const useUserStore = create<UserState>()(
  persist(
    (set, get) => ({
      token: null,
      userInfo: null,
      isLoggedIn: false,
      level: 1,
      points: 0,

      login: async (username, password) => {
        const res = await login({ username, password })
        set({
          token: res.token,
          userInfo: res.user,
          isLoggedIn: true,
          level: res.user.level,
          points: res.user.points
        })
      },

      loginByCode: async (phone, code) => {
        const res = await loginBySMS({ phone, code })
        set({
          token: res.token,
          userInfo: res.user,
          isLoggedIn: true
        })
      },

      socialLogin: async (provider) => {
        // OAuth 跳转
        const oauthUrl = getOAuthUrl(provider)
        window.location.href = oauthUrl
      },

      logout: () => {
        set({ token: null, userInfo: null, isLoggedIn: false, level: 1, points: 0 })
        cartStore.getState().clearSelected()
      },

      updateProfile: async (data) => {
        const updated = await updateUserProfile(data)
        set({ userInfo: updated })
      },

      addPoints: (points, reason) => {
        set(state => ({
          points: state.points + points,
          level: calculateLevel(state.points + points)
        }))
        // 记录积分变动
        logPointsChange({ points, reason, balance: get().points })
      }
    }),
    { name: 'user-storage' }
  )
)

// 加入购物车动画
const animateAddToCart = (itemId: string) => {
  // 1. 商品图片飞入购物车
  const imgRect = document.getElementById(`product-${itemId}`)?.getBoundingClientRect()
  const cartRect = document.getElementById('tab-cart')?.getBoundingClientRect()

  // 2. 创建动画元素
  const flyingImg = document.createElement('img')
  flyingImg.src = getProductImage(itemId)
  flyingImg.className = 'fixed z-50 w-12 h-12 rounded-full object-cover'
  flyingImg.style.left = `${imgRect?.left}px`
  flyingImg.style.top = `${imgRect?.top}px`

  document.body.appendChild(flyingImg)

  // 3. 动画轨迹
  animate(flyingImg).to(cartRect.left, cartRect.top, 300)
    .scale(0.3)
    .ease('ease-out')
    .onComplete(() => {
      flyingImg.remove()
      // 购物车弹跳动画
      const cartEl = document.getElementById('tab-cart')
      cartEl?.classList.add('animate-bounce')
      setTimeout(() => cartEl?.classList.remove('animate-bounce'), 500)
    })
}

### 4.3 响应式设计

```typescript
// hooks/useResponsive.ts
export const useResponsive = () => {
  const [device, setDevice] = useState<'mobile' | 'tablet' | 'desktop'>('desktop')

  useEffect(() => {
    const updateDevice = () => {
      const width = window.innerWidth
      if (width < 768) setDevice('mobile')
      else if (width < 1024) setDevice('tablet')
      else setDevice('desktop')
    }

    updateDevice()
    window.addEventListener('resize', updateDevice)
    return () => window.removeEventListener('resize', updateDevice)
  }, [])

  return {
    isMobile: device === 'mobile',
    isTablet: device === 'tablet',
    isDesktop: device === 'desktop',
    device
  }
}

// 布局组件自动切换
const ResponsiveLayout = ({ children }) => {
  const { isMobile } = useResponsive()
  return isMobile ? <MobileLayout>{children}</MobileLayout> : <DesktopLayout>{children}</DesktopLayout>
}
```

import { FixedSizeGrid } from 'react-virtualized'

const ProductGrid = ({ products }: { products: Product[] }) => {

  const rowCount = Math.ceil(products.length / columnCount)

  return (
    <FixedSizeGrid

      rowHeight={CARD_HEIGHT + 16}

      {({ columnIndex, rowIndex, style }) => {

        const product = products[index]
        if (!product) return null
        return (
          <div style={style}>
            <ProductCard product={product} />
          </div>
        )
      }}
    </FixedSizeGrid>
  )
}
